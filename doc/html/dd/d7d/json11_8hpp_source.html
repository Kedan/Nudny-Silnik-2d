267 static void dump(const string &value, string &out) {
268 out += '"';
269 for (size_t i = 0; i < value.length(); i++) {
270 const char ch = value[i];
271 if (ch == '\\') {
272 out += "\\\\";
273 } else if (ch == '"') {
274 out += "\\\"";
275 } else if (ch == '\b') {
276 out += "\\b";
277 } else if (ch == '\f') {
278 out += "\\f";
279 } else if (ch == '\n') {
280 out += "\\n";
281 } else if (ch == '\r') {
282 out += "\\r";
283 } else if (ch == '\t') {
284 out += "\\t";
285 } else if (static_cast<uint8_t>(ch) <= 0x1f) {
286 char buf[8];
287 snprintf(buf, sizeof buf, "\\u%04x", ch);
288 out += buf;
289 } else if (static_cast<uint8_t>(ch) == 0xe2 && static_cast<uint8_t>(value[i+1]) == 0x80
290 && static_cast<uint8_t>(value[i+2]) == 0xa8) {
291 out += "\\u2028";
292 i += 2;
293 } else if (static_cast<uint8_t>(ch) == 0xe2 && static_cast<uint8_t>(value[i+1]) == 0x80
294 && static_cast<uint8_t>(value[i+2]) == 0xa9) {
295 out += "\\u2029";
296 i += 2;
297 } else {
298 out += ch;
299 }
300 }
301 out += '"';
302 }

326 }
327 out += "}";

547 /* fail(msg, err_ret = Json())
548 *
549 * Mark this parse as failed.

551 Json fail(string &&msg) {

555 template <typename T>
556 T fail(string &&msg, const T err_ret) {
557 if (!failed)
558 err = std::move(msg);
559 failed = true;
560 return err_ret;
561 }

563 /* consume_whitespace()
564 *
565 * Advance until the current character is non-whitespace.
566 */
567 void consume_whitespace() {
568 while (str[i] == ' ' || str[i] == '\r' || str[i] == '\n' || str[i] == '\t')
569 i++;
570 }
571
572 /* consume_comment()
573 *
574 * Advance comments (c-style inline and multiline).
575 */
576 bool consume_comment() {
577 bool comment_found = false;
578 if (str[i] == '/') {
579 i++;
580 if (i == str.size())
581 return fail("unexpected end of input after start of comment", false);
582 if (str[i] == '/') { // inline comment
583 i++;
584 // advance until next line, or end of input
585 while (i < str.size() && str[i] != '\n') {
586 i++;
587 }
588 comment_found = true;
589 }
590 else if (str[i] == '*') { // multiline comment
591 i++;
592 if (i > str.size()-2)
593 return fail("unexpected end of input inside multi-line comment", false);
594 // advance until closing tokens
595 while (!(str[i] == '*' && str[i+1] == '/')) {
596 i++;
597 if (i > str.size()-2)
598 return fail(
599 "unexpected end of input inside multi-line comment", false);
600 }
601 i += 2;
602 comment_found = true;
603 }
604 else
605 return fail("malformed comment", false);
606 }
607 return comment_found;
608 }
609
610 /* consume_garbage()
611 *
612 * Advance until the current character is non-whitespace and non-comment.
613 */
614 void consume_garbage() {
615 consume_whitespace();
616 if(strategy == JsonParse::COMMENTS) {
617 bool comment_found = false;
618 do {
619 comment_found = consume_comment();
620 if (failed) return;
621 consume_whitespace();
622 }
623 while(comment_found);
624 }
625 }
626
627 /* get_next_token()
628 *
629 * Return the next non-whitespace character. If the end of the input is reached,
630 * flag an error and return 0.
631 */
632 char get_next_token() {
633 consume_garbage();
634 if (failed) return static_cast<char>(0);
635 if (i == str.size())
636 return fail("unexpected end of input", static_cast<char>(0));
637
638 return str[i++];
639 }

641 /* encode_utf8(pt, out)
642 *
643 * Encode pt as UTF-8 and add it to out.
644 */
645 void encode_utf8(long pt, string & out) {
646 if (pt < 0)
647 return;
648
649 if (pt < 0x80) {
650 out += static_cast<char>(pt);
651 } else if (pt < 0x800) {
652 out += static_cast<char>((pt >> 6) | 0xC0);
653 out += static_cast<char>((pt & 0x3F) | 0x80);
654 } else if (pt < 0x10000) {
655 out += static_cast<char>((pt >> 12) | 0xE0);
656 out += static_cast<char>(((pt >> 6) & 0x3F) | 0x80);
657 out += static_cast<char>((pt & 0x3F) | 0x80);
658 } else {
659 out += static_cast<char>((pt >> 18) | 0xF0);
660 out += static_cast<char>(((pt >> 12) & 0x3F) | 0x80);
661 out += static_cast<char>(((pt >> 6) & 0x3F) | 0x80);
662 out += static_cast<char>((pt & 0x3F) | 0x80);
663 }
664 }

666 /* parse_string()
667 *
668 * Parse a string, starting at the current position.
669 */
670 string parse_string() {
671 string out;
672 long last_escaped_codepoint = -1;
673 while (true) {
674 if (i == str.size())
675 return fail("unexpected end of input in string", "");
676
677 char ch = str[i++];
678
679 if (ch == '"') {
680 encode_utf8(last_escaped_codepoint, out);
681 return out;
682 }
683
684 if (in_range(ch, 0, 0x1f))
685 return fail("unescaped " + esc(ch) + " in string", "");
686
687 // The usual case: non-escaped characters
688 if (ch != '\\') {
689 encode_utf8(last_escaped_codepoint, out);
690 last_escaped_codepoint = -1;
691 out += ch;
692 continue;
693 }
694
695 // Handle escapes
696 if (i == str.size())
697 return fail("unexpected end of input in string", "");
698
699 ch = str[i++];
700
701 if (ch == 'u') {
702 // Extract 4-byte escape sequence
703 string esc = str.substr(i, 4);
704 // Explicitly check length of the substring. The following loop
705 // relies on std::string returning the terminating NUL when
706 // accessing str[length]. Checking here reduces brittleness.
707 if (esc.length() < 4) {
708 return fail("bad \\u escape: " + esc, "");
709 }
710 for (size_t j = 0; j < 4; j++) {
711 if (!in_range(esc[j], 'a', 'f') && !in_range(esc[j], 'A', 'F')
712 && !in_range(esc[j], '0', '9'))
713 return fail("bad \\u escape: " + esc, "");
714 }
715
716 long codepoint = strtol(esc.data(), nullptr, 16);
717
718 // JSON specifies that characters outside the BMP shall be encoded as a pair
719 // of 4-hex-digit \u escapes encoding their surrogate pair components. Check
720 // whether we're in the middle of such a beast: the previous codepoint was an
721 // escaped lead (high) surrogate, and this is a trail (low) surrogate.
722 if (in_range(last_escaped_codepoint, 0xD800, 0xDBFF)
723 && in_range(codepoint, 0xDC00, 0xDFFF)) {
724 // Reassemble the two surrogate pairs into one astral-plane character, per
725 // the UTF-16 algorithm.
726 encode_utf8((((last_escaped_codepoint - 0xD800) << 10)
727 | (codepoint - 0xDC00)) + 0x10000, out);
728 last_escaped_codepoint = -1;
729 } else {
730 encode_utf8(last_escaped_codepoint, out);
731 last_escaped_codepoint = codepoint;
732 }
733
734 i += 4;
735 continue;
736 }
737
738 encode_utf8(last_escaped_codepoint, out);
739 last_escaped_codepoint = -1;
740
741 if (ch == 'b') {
742 out += '\b';
743 } else if (ch == 'f') {
744 out += '\f';
745 } else if (ch == 'n') {
746 out += '\n';
747 } else if (ch == 'r') {
748 out += '\r';
749 } else if (ch == 't') {
750 out += '\t';
751 } else if (ch == '"' || ch == '\\' || ch == '/') {
752 out += ch;
753 } else {
754 return fail("invalid escape character " + esc(ch), "");
755 }
756 }
757 }

759 /* parse_number()
760 *
761 * Parse a double.
762 */
763 Json parse_number() {
764 size_t start_pos = i;
765
766 if (str[i] == '-')
767 i++;
768
769 // Integer part
770 if (str[i] == '0') {
771 i++;
772 if (in_range(str[i], '0', '9'))
773 return fail("leading 0s not permitted in numbers");
774 } else if (in_range(str[i], '1', '9')) {
775 i++;
776 while (in_range(str[i], '0', '9'))
777 i++;
778 } else {
779 return fail("invalid " + esc(str[i]) + " in number");
780 }
781
782 if (str[i] != '.' && str[i] != 'e' && str[i] != 'E'
783 && (i - start_pos) <= static_cast<size_t>(std::numeric_limits<int>::digits10)) {
784 return std::atoi(str.c_str() + start_pos);
785 }
786
787 // Decimal part
788 if (str[i] == '.') {
789 i++;
790 if (!in_range(str[i], '0', '9'))
791 return fail("at least one digit required in fractional part");
792
793 while (in_range(str[i], '0', '9'))
794 i++;
795 }
796
797 // Exponent part
798 if (str[i] == 'e' || str[i] == 'E') {
799 i++;
800
801 if (str[i] == '+' || str[i] == '-')
802 i++;
803
804 if (!in_range(str[i], '0', '9'))
805 return fail("at least one digit required in exponent");
806
807 while (in_range(str[i], '0', '9'))
808 i++;
809 }
810
811 return std::strtod(str.c_str() + start_pos, nullptr);
812 }

830 /* parse_json()
831 *
832 * Parse a JSON object.
833 */
834 Json parse_json(int depth) {
835 if (depth > max_depth) {
836 return fail("exceeded maximum nesting depth");
837 }
838
839 char ch = get_next_token();
840 if (failed)
841 return Json();
842
843 if (ch == '-' || (ch >= '0' && ch <= '9')) {
844 i--;
845 return parse_number();
846 }
847
848 if (ch == 't')
849 return expect("true", true);
850
851 if (ch == 'f')
852 return expect("false", false);
853
854 if (ch == 'n')
855 return expect("null", Json());
856
857 if (ch == '"')
858 return parse_string();
859
860 if (ch == '{') {
861 map<string, Json> data;
862 ch = get_next_token();
863 if (ch == '}')
864 return data;
865
866 while (1) {
867 if (ch != '"')
868 return fail("expected '\"' in object, got " + esc(ch));
869
870 string key = parse_string();
871 if (failed)
872 return Json();
873
874 ch = get_next_token();
875 if (ch != ':')
876 return fail("expected ':' in object, got " + esc(ch));
877
878 data[std::move(key)] = parse_json(depth + 1);
879 if (failed)
880 return Json();
881
882 ch = get_next_token();
883 if (ch == '}')
884 break;
885 if (ch != ',')
886 return fail("expected ',' in object, got " + esc(ch));
887
888 ch = get_next_token();
889 }
890 return data;
891 }
892
893 if (ch == '[') {
894 vector<Json> data;
895 ch = get_next_token();
896 if (ch == ']')
897 return data;
898
899 while (1) {
900 i--;
901 data.push_back(parse_json(depth + 1));
902 if (failed)
903 return Json();
904
905 ch = get_next_token();
906 if (ch == ']')
907 break;
908 if (ch != ',')
909 return fail("expected ',' in list, got " + esc(ch));
910
911 ch = get_next_token();
912 (void)ch;
913 }
914 return data;
915 }
916
917 return fail("expected value, got " + esc(ch));
918 }

924 Json result = parser.parse_json(0);

927 parser.consume_garbage();
928 if (parser.failed)

931 ((parser.i + 1) != in.size() && in[parser.i] != 0)) //RBP: If there is only 1 character diff, it is probably just a terminating zero from a memory read.
