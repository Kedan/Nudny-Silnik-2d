42bool tson::Tile::parse(IJson &json, tson::Tileset *tileset, tson::Map *map)
43{
44 m_tileset = tileset;
45 m_map = map;
46
47 if(json.count("image") > 0) m_image = fs::path(json["image"].get<std::string>()); //Optional
48
49 bool allFound = parseId(json);
50
51 if(json.count("type") > 0) m_type = json["type"].get<std::string>(); //Optional
52 else if(json.count("class") > 0) m_type = json["class"].get<std::string>(); //Tiled v1.9 renamed 'type' to 'class'
53
54 if(json.count("objectgroup") > 0) m_objectgroup = tson::Layer(json["objectgroup"], m_map); //Optional
55
56 if(json.count("imagewidth") > 0 && json.count("imageheight") > 0)
57 m_imageSize = {json["imagewidth"].get<int>(), json["imageheight"].get<int>()}; //Optional
58
59 m_subRect = {0,0, m_imageSize.x, m_imageSize.y};
60 if(json.count("x") > 0) m_subRect.x = json["x"].get<int>(); //Optional
61 if(json.count("y") > 0) m_subRect.y = json["y"].get<int>(); //Optional
62 if(json.count("width") > 0) m_subRect.width = json["width"].get<int>(); //Optional
63 if(json.count("height") > 0) m_subRect.height = json["height"].get<int>(); //Optional
64
65 //More advanced data
66 if(json.count("animation") > 0 && json["animation"].isArray())
67 {
68 auto &animation = json.array("animation");
69 std::vector<tson::Frame> frames;
70 std::for_each(animation.begin(), animation.end(), [&](std::unique_ptr<IJson> &item) { frames.emplace_back(*item); });
71 if(frames.size() > 0)
72 {
73 m_animation.setFrames(frames);
74 }
75 }
76 if(json.count("terrain") > 0 && json["terrain"].isArray())
77 {
78 auto &terrain = json.array("terrain");
79 std::for_each(terrain.begin(), terrain.end(), [&](std::unique_ptr<IJson> &item) { m_terrain.emplace_back(item->get<int>()); });
80 }
81
82 if(json.count("properties") > 0 && json["properties"].isArray())
83 {
84 auto &properties = json.array("properties");
85 tson::Project *project = (m_map != nullptr) ? m_map->getProject() : nullptr;
86 std::for_each(properties.begin(), properties.end(), [&](std::unique_ptr<IJson> &item) { m_properties.add(*item, project); });
87 }
88
89 performDataCalculations();
90
91 return allFound;
92}

125void tson::Tile::performDataCalculations()
126{
127 if(m_tileset == nullptr || m_map == nullptr)
128 return;
129
130 int firstId = m_tileset->getFirstgid(); //First tile id of the tileset
131 int columns = m_tileset->getColumns();
132 int rows = m_tileset->getTileCount() / columns;
133 int lastId = (m_tileset->getFirstgid() + m_tileset->getTileCount()) - 1;
134
135 int const gid = static_cast<int>(getGid());
136 if (gid >= firstId && gid <= lastId)
137 {
138 int const baseTilePosition = (gid - firstId);
139
140 int const tileModX = (baseTilePosition % columns);
141 int const currentRow = (baseTilePosition / columns);
142 int const offsetX = (tileModX != 0) ? ((tileModX) * m_map->getTileSize().x) : (0 * m_map->getTileSize().x);
143 int const offsetY = (currentRow < rows-1) ? (currentRow * m_map->getTileSize().y) : ((rows-1) * m_map->getTileSize().y);
144
145 tson::Vector2i spacing = m_tileset->getMarginSpacingOffset({tileModX, currentRow});
146 m_drawingRect = { offsetX + spacing.x, offsetY + spacing.y, m_tileset->getTileSize().x, m_tileset->getTileSize().y };
147 }
148 else
149 m_drawingRect = {0, 0, 0, 0};
150}

228void tson::Layer::decompressData()
229{
230
231 tson::DecompressorContainer *container = m_map->getDecompressors();
232 if(container->empty())
233 return;
234
235 if(m_encoding.empty() && m_compression.empty())
236 return;
237
238 std::string data = m_base64Data;
239 bool hasBeenDecoded = false;
240 if(!m_encoding.empty() && container->contains(m_encoding))
241 {
242 data = container->get(m_encoding)->decompress(data);
243 hasBeenDecoded = true;
244 }
245
246 if(!m_compression.empty() && container->contains(m_compression))
247 {
248 data = container->get(m_compression)->decompress(data);
249 }
250
251 if(hasBeenDecoded)
252 {
253 std::vector<uint8_t> bytes = tson::Tools::Base64DecodedStringToBytes(data);
254 m_data = tson::Tools::BytesToUnsignedInts(bytes);
255 }
256}

465void tson::readGid(uint32_t& gid, TileFlipFlags& flags, IJson& main, IJson* templ)
466{
467 uint32_t g;
468 if(readField(g, "gid", main, templ))
469 {
470 if (g & FLIPPED_HORIZONTALLY_FLAG) flags |= TileFlipFlags::Horizontally;
471 if (g & FLIPPED_VERTICALLY_FLAG) flags |= TileFlipFlags::Vertically;
472 if (g & FLIPPED_DIAGONALLY_FLAG) flags |= TileFlipFlags::Diagonally;
473
474 // Clear flags
475 g &= ~(FLIPPED_HORIZONTALLY_FLAG | FLIPPED_VERTICALLY_FLAG | FLIPPED_DIAGONALLY_FLAG);
476
477 gid = g;
478 }

488bool tson::Object::parse(IJson &json, tson::Map *map)
489{
490 m_map = map;
491 bool allFound = true;
492 IJson* templateJson = nullptr;
493
494 if(readField(m_template, "template", json) && map != nullptr)
495 {
496 IJson* tobjJsonFile = map->parseLinkedFile(m_template);
497 if(tobjJsonFile)
498 templateJson = readField("object", *tobjJsonFile);
499 }
500
501 readField(m_ellipse, "ellipse", json, templateJson); //Optional
502 readField(m_point, "point", json, templateJson); // Optional
503 readField(m_text, "text", json, templateJson);
504 readGid(m_gid, m_flipFlags, json, templateJson);
505
506 allFound &= readField(m_id, "id", json, templateJson);
507 allFound &= readField(m_name, "name", json, templateJson);
508 allFound &= readField(m_rotation, "rotation", json, templateJson);
509 allFound &= readField(m_type, "type", json, templateJson) || readField(m_type, "class", json, templateJson); //Tiled v1.9 renamed 'type' to 'class'
510 allFound &= readField(m_visible, "visible", json, templateJson);
511 allFound &= readVector(m_size, "width", "height", json, templateJson);
512 allFound &= readVector(m_position, "x", "y", json, templateJson);
513
514 setObjectTypeByJson(json, templateJson);
515
516 if(m_objectType == ObjectType::Template)
517 allFound = true; //Just accept anything with this type
518
519 //More advanced data
520 readField(m_polygon, "polygon", json, templateJson);
521 readField(m_polyline, "polyline", json, templateJson);
522
523 // merge properties with template's.
524 if(templateJson) readProperties(m_properties, *templateJson, m_map);
525 readProperties(m_properties, json, m_map);
526
527 return allFound;
528}

607void tson::Property::setValueByType(IJson &json)
608{
609 switch(m_type)
610 {
611 case Type::Color:
612 m_value = Colori(json.get<std::string>());
613 break;
614
615 case Type::File:
616 m_value = fs::path(json.get<std::string>());
617 break;
618
619 case Type::Int:
620 if(!m_propertyType.empty())
621 {
622 m_type = Type::Enum;
623 tson::EnumDefinition *def = (m_project != nullptr) ? m_project->getEnumDefinition(m_propertyType) : nullptr;
624 if(def != nullptr)
625 {
626 uint32_t v = json.get<uint32_t>();
627 m_value = tson::EnumValue(v, def);
628 }
629 else
630 m_value = tson::EnumValue();
631 }
632 else
633 m_value = json.get<int>();
634
635 break;
636
637 case Type::Boolean:
638 m_value = json.get<bool>();
639 break;
640
641 case Type::Float:
642 m_value = json.get<float>();
643 break;
644
645 case Type::String:
646 if(!m_propertyType.empty())
647 {
648 m_type = Type::Enum;
649 tson::EnumDefinition *def = (m_project != nullptr) ? m_project->getEnumDefinition(m_propertyType) : nullptr;
650 if(def != nullptr)
651 {
652 std::string v = json.get<std::string>();
653 m_value = tson::EnumValue(v, def);
654 }
655 else
656 m_value = tson::EnumValue();
657 }
658 else
659 setStrValue(json.get<std::string>());
660
661 break;
662
663 case Type::Class:
664 {
665 tson::TiledClass *baseClass = (m_project != nullptr) ? m_project->getClass(m_propertyType) : nullptr;
666 if (baseClass != nullptr)
667 {
668 tson::TiledClass c = *baseClass;
669 c.update(json);
670 m_value = c;
671 }
672 }
673 break;
674
675 case Type::Object:
676 m_value = json.get<uint32_t>();
677 break;
678 default:
679 setStrValue(json.get<std::string>());
680 break;
681 }
682}
